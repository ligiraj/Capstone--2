Convert non-numerical variables into numerical variables¶

In [29]:
import pandas as pd
Df_insurance = pd.read_csv("insurance (1).csv")

Df_insurance["sex"] = [0 if i == "female" else 1 for i in Df_insurance["sex"]]
Df_insurance["sex"] = Df_insurance["sex"].astype("category")
Df_insurance = pd.get_dummies(Df_insurance, columns= ["sex"])
Df_insurance["smoker"]=[0 if i=="no" else 1 for i in Df_insurance["smoker"]]
Df_insurance["smoker"]=Df_insurance["smoker"].astype("category")
Df_insurance = pd.get_dummies(Df_insurance , columns=["smoker"])
Df_insurance["region"] = [0 if i == "southeast" else 1 if i == "southwest" else 2 if i == "northwest" else 3 for i in Df_insurance["region"]] 
Df_insurance["region"]=Df_insurance["region"].astype("category")
Df_insurance = pd.get_dummies(Df_insurance , columns=["region"])
